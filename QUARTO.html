# Define global parameters and number of replications
# =====================================================================
R <- 100000       # Number of simulations
beta0 <- 10       # True intercept
beta1 <- 1        # True slope
n_vals <- c(25, 50, 100)
# =====================================================================
# Part a) Data Generation & Average x
# Set seed, simulate x values once, and report averages
# =====================================================================
set.seed(549)

# Generate and store x values in a list so they remain fixed
x_list <- list()
for (n in n_vals) {
  x_list[[as.character(n)]] <- rnorm(n, mean = 0, sd = 1)
}

# Calculate and print the average of x for each sample size
x_averages <- sapply(x_list, mean)
print("Average x values for n = 25, 50, 100:")

# =====================================================================
# Part b & c) Baseline Simulation (n=25, gamma=0)
# =====================================================================
n <- 25
gamma <- 0
x <- x_list[[as.character(n)]]

# 1. Calculate standard deviations (sqrt of variance)
sd_eps <- sqrt(exp(gamma * x))

# 2. Simulate errors (100,000 columns, 25 rows)
eps_mat <- matrix(rnorm(n * R, mean = 0, sd = sd_eps), nrow = n, ncol = R)

# 3. Compute responses Y
y_mat <- beta0 + beta1 * x + eps_mat

# 4. Fast Matrix OLS: beta_hat = (X'X)^(-1) X' Y
X_mat <- cbind(1, x)
H <- solve(t(X_mat) %*% X_mat) %*% t(X_mat)  # Projection matrix
beta_hats <- H %*% y_mat                     # Computes 100,000 regressions instantly

# Calculate and print averages
avg_beta0 <- mean(beta_hats[1, ])
avg_beta1 <- mean(beta_hats[2, ])

# =====================================================================
# Part e) Autoregression Model Simulation
# Loop through all combinations of rho and n
# =====================================================================
rho_vals <- c(0, 0.5, 0.9)
results_ar <- expand.grid(n = n_vals, rho = rho_vals)
results_ar$avg_beta0 <- NA
results_ar$avg_beta1 <- NA

for (i in 1:nrow(results_ar)) {
  n <- results_ar$n[i]
  rho <- results_ar$rho[i]
  x <- x_list[[as.character(n)]]
  
  # Initialize error matrix
  eps_mat <- matrix(0, nrow = n, ncol = R)
  eps_mat[1, ] <- 3  # Set epsilon_1 = 3 for all 100,000 paths
  
  # Generate standard normal u_i
  u_mat <- matrix(rnorm((n - 1) * R, mean = 0, sd = 1), nrow = n - 1, ncol = R)
  
  # Build the autoregressive series iteratively
  for (j in 2:n) {
    eps_mat[j, ] <- rho * eps_mat[j - 1, ] + u_mat[j - 1, ]
  }
  
  y_mat <- beta0 + beta1 * x + eps_mat
  
  X_mat <- cbind(1, x)
  H <- solve(t(X_mat) %*% X_mat) %*% t(X_mat)
  beta_hats <- H %*% y_mat
  
  results_ar$avg_beta0[i] <- mean(beta_hats[1, ])
  results_ar$avg_beta1[i] <- mean(beta_hats[2, ])
}
